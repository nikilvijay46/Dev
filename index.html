Program 7 (Apache2-install.yml)

---
- name: install apache2 on client servers
    hosts: all
    become: yes
    gather_facts: true
        
    vars:
    web_server: apache2
        
    tasks:
    - name: update cache
        apt: 
        update_cache: yes
        
    - name: install apache2
        apt: 
        name: "{{ web_server }}"
        state: present
        
    - name: copy index.html to web root
        copy:
        src: index.html
        dest: /var/www/html/
        
    handlers:
    - name: Restart apache2
        service:
        name: apache2
        state: restarted
